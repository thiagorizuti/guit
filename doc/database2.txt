class user{
    primary key int id;
    string[20] name;
    string[20] lastname;
    string[20] phonenumber;
    string[20] username;
    string[20] password;
    string[30] email;
    string[30] specialization;
    date lastaccess;
    date joindate;
    boolean isadmin;

}

class badge{
    primary key int id;
    string[40] url;
    string[20] category;
    string[100] about;
    int points;
}

class user_earns_badge{
    primary key int id;
    user user;
    badge badge;
}

class teachingpractice{
    primary key int id;
    user author;
    string[20] title;
    datetime time;
    memo problemstatement;
    memo thisbundle;
    memo wayitworks;
    memo worksbetter;
    memo doesntwork;
    memo doesntworkunless;
    memo workedif;
    memo variations;
    memo solutionstatement;
}

class user_comments_teachingpractice{
    primary key int id;
    user user;
    teachingpractice teachingpractice;
    datetime time;
    memo comment;
    user_comments_teachingpractice reply;
}

class user_likes_teachingpractice{
    primary key int id;
    user user;
    teachingpractice teachingpractice;
    datetime time;
}
