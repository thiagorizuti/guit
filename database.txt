class user{
    primary key int id;
    string[20] name;
    string[20] lastname;
    string[20] phonenumber;
    string[20] username;
    string[20] password;
    string[20] email;
    string[20] specialization;
    int points;
    date lastaccess;
    date joindate;
    boolean isadmin;
}

class admin{
    primary key int id;
    string[20] role;
}

class badge{
    primary key int id;
    string[20] url;
}

class user_earns_badge{
    primary key int id;
    user user;

}

    primary key int id;
    user user;
    user recipiant;
    memo content;

}
